#include<stdio.h>
int main()
{
    int n,first,last,swap;
    printf("Enter the number: ");
    scanf("%d",&n);

    last= n%10;
    
    first=n;
    
    while(first>=10)
    {
        first=first / 10;
    }
    swap = (last*10)+first;

    printf("First digit = %d\n", first);
    printf("Last digit = %d\n", last);
    printf("Swapped=%d",swap);

    return 0;

}
